DROP TABLE IF EXISTS \`tblLoanDetail\`;

CREATE TABLE \`tblSystemUser\` (  
  \`userID\` VARCHAR(20) NOT NULL,  
  \`username\` VARCHAR(50) NOT NULL,  
  \`role\` VARCHAR(50) NOT NULL,  
  \`password\` VARCHAR(50) NOT NULL,  
  \`fullName\` VARCHAR(50) NOT NULL,  
  \`email\` VARCHAR(50) NOT NULL,  
  \`status\` VARCHAR(20) NOT NULL,  
  \`createdDate\` DATE NOT NULL,  
  \`lockReason\` VARCHAR(255) NULL,  
  \`lockUntil\` DATE NULL,  
  PRIMARY KEY (\`userID\`)  
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4\_unicode\_ci;

CREATE TABLE \`tblPatron\` (  
  \`patronID\` VARCHAR(20) NOT NULL,  
  \`name\` VARCHAR(50) NOT NULL,  
  \`phone\` VARCHAR(15) NOT NULL,  
  \`email\` VARCHAR(50) NOT NULL,  
  \`cardType\` VARCHAR(20) NOT NULL,  
  \`expiryDate\` DATE NOT NULL,  
  \`username\` VARCHAR(50) NOT NULL,  
  \`password\` VARCHAR(50) NOT NULL,  
  PRIMARY KEY (\`patronID\`)  
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4\_unicode\_ci;

CREATE TABLE \`tblShelf\` (  
  \`shelfID\` VARCHAR(20) NOT NULL,  
  \`room\` VARCHAR(50) NOT NULL,  
  \`row\` VARCHAR(50) NOT NULL,  
  PRIMARY KEY (\`shelfID\`)  
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4\_unicode\_ci;

CREATE TABLE \`tblBook\` (  
  \`bookID\` VARCHAR(20) NOT NULL,  
  \`title\` VARCHAR(100) NOT NULL,  
  \`isbn\` VARCHAR(20) NOT NULL,  
  \`publishYear\` INT NOT NULL,  
  \`deDCcode\` VARCHAR(20) NOT NULL,  
  \`price\` DOUBLE NOT NULL,  
  \`summary\` VARCHAR(255) NOT NULL,  
  PRIMARY KEY (\`bookID\`)  
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4\_unicode\_ci;

CREATE TABLE \`tblItem\` (  
  \`itemID\` VARCHAR(20) NOT NULL,  
  \`barCode\` VARCHAR(50) NOT NULL,  
  \`status\` VARCHAR(20) NOT NULL,  
  \`tblBookID\` VARCHAR(20) NOT NULL,  
  \`tblShelfID\` VARCHAR(20) NOT NULL,  
  PRIMARY KEY (\`itemID\`),  
  CONSTRAINT \`fk\_tblItem\_tblBook\` FOREIGN KEY (\`tblBookID\`) REFERENCES \`tblBook\` (\`bookID\`),  
  CONSTRAINT \`fk\_tblItem\_tblShelf\` FOREIGN KEY (\`tblShelfID\`) REFERENCES \`tblShelf\` (\`shelfID\`)  
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4\_unicode\_ci;

CREATE TABLE \`tblLoanRecord\` (  
  \`loanID\` VARCHAR(20) NOT NULL,  
  \`loanDate\` DATE NOT NULL,  
  \`dueDate\` DATE NOT NULL,  
  \`fineAmount\` DOUBLE NOT NULL,  
  \`tblSystemUserID\` VARCHAR(20) NOT NULL,  
  \`tblPatronID\` VARCHAR(20) NOT NULL,  
  PRIMARY KEY (\`loanID\`),  
  CONSTRAINT \`fk\_tblLoanRecord\_tblSystemUser\` FOREIGN KEY (\`tblSystemUserID\`) REFERENCES \`tblSystemUser\` (\`userID\`),  
  CONSTRAINT \`fk\_tblLoanRecord\_tblPatron\` FOREIGN KEY (\`tblPatronID\`) REFERENCES \`tblPatron\` (\`patronID\`)  
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4\_unicode\_ci;

CREATE TABLE \`tblLoanDetail\` (  
  \`tblLoanRecordID\` VARCHAR(20) NOT NULL,  
  \`tblItemID\` VARCHAR(20) NOT NULL,  
  \`returnDate\` DATE NULL,  
  \`conditionNote\` VARCHAR(255) NOT NULL,  
  PRIMARY KEY (\`tblLoanRecordID\`, \`tblItemID\`),  
  CONSTRAINT \`fk\_tblLoanDetail\_tblLoanRecord\` FOREIGN KEY (\`tblLoanRecordID\`) REFERENCES \`tblLoanRecord\` (\`loanID\`),  
  CONSTRAINT \`fk\_tblLoanDetail\_tblItem\` FOREIGN KEY (\`tblItemID\`) REFERENCES \`tblItem\` (\`itemID\`)  
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4\_unicode\_ci;

CREATE TABLE \`tblReservation\` (  
  \`reservID\` VARCHAR(20) NOT NULL,  
  \`resDate\` DATE NOT NULL,  
  \`status\` VARCHAR(20) NOT NULL,  
  \`tblPatronID\` VARCHAR(20) NOT NULL,  
  \`tblBookID\` VARCHAR(20) NOT NULL,  
  PRIMARY KEY (\`reservID\`),  
  CONSTRAINT \`fk\_tblReservation\_tblPatron\` FOREIGN KEY (\`tblPatronID\`) REFERENCES \`tblPatron\` (\`patronID\`),  
  CONSTRAINT \`fk\_tblReservation\_tblBook\` FOREIGN KEY (\`tblBookID\`) REFERENCES \`tblBook\` (\`bookID\`)  
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4\_unicode\_ci;
